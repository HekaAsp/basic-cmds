automatisch starten 

docker update --restart unless-stopped open-webui 
docker start open-webui


server starten 

docker run -d \
  --name ollama \
  --restart unless-stopped \
  --gpus all \
  -p 11434:11434 \
  -v ollama:/root/.ollama \
  ollama/ollama

docker update --restart unless-stopped ollama
docker start ollama

bridge anlegen 
docker network create ai

web ui starten
docker run -d \
  --name open-webui \
  --network ai \
  --restart unless-stopped \
  -p 3000:8080 \
  -e OLLAMA_BASE_URL=http://ollama:11434 \
  -e OLLAMA_MODEL=deepseek-r1:1.5b \
  ghcr.io/open-webui/open-webui:ollama
---------------------------------------------------------
prüfen ob web ui ollama-server findet 
docker exec -it open-webui2 curl http://ollama:11434/api/tags
